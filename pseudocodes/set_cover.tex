\begin{algorithm}[h]
\DontPrintSemicolon
\LinesNumbered
\caption{The $\gamma$-greedy algorithm for PCSC}
\label{PCSCalgorithm}
\SetKwProg{Proc}{procedure}{}{}
\Proc{\textsc{PCSC}$(\mathcal{U}, \mathcal{S}, \mathcal{F}, K)$}{
$\mathcal{C} \gets \emptyset$\;

\While{$\spr{\cov\br{\mathcal{C}, \mathcal{U}}} < K$}{
$\mathcal{A} \gets$ Run the $\gamma$-approx. algorithm for MDPCS on $(\mathcal{U} - \mathcal{C}, \mathcal{S}-\mathcal{C}, \mathcal{F}-\mathcal{C}, m)$\;

\If{$\spr{\cov\br{\mathcal{C}\cup \mathcal{A}, \mathcal{U}}} \geq K$}
{
    Find the minimum budget $B \in [\spr{\mathcal{A}}]$, such that the $\gamma$-approx. algorithm for MDPCS on $(\mathcal{U} - \mathcal{C}, \mathcal{S}-\mathcal{C}, \mathcal{F}-\mathcal{C}, B)$ returns a set 
    $\mathcal{B}$ with $\spr{\cov\br{\mathcal{B}, \mathcal{U}-\mathcal{C}}} \geq \frac{K-\cov\br{\mathcal{C}, \mathcal{U}}}{\alpha}$\;

    \While{$\spr{\cov\br{\mathcal{C}, \mathcal{U}}} < K$}{
        $\mathcal{B} \gets$ Run the $\gamma$-approx. algorithm for MDPCS on $(\mathcal{U} - \mathcal{C}, \mathcal{S}-\mathcal{C}, \mathcal{F}-\mathcal{C}, B)$\;
        
        $\mathcal{C}\gets \mathcal{C} \cup \mathcal{B}$\;
    }

    \Return $\mathcal{C}$
}
\ForEach{$u \in \cov\br{\mathcal{A}, \mathcal{U}-\mathcal{C}}$}{
    $c\br{u}\gets \Delta\br{\mathcal{A}, \mathcal{U}-\mathcal{C}}$\;
}
$\mathcal{C} \gets \mathcal{C} \cup \mathcal{A}$\;
}
}
\end{algorithm}
