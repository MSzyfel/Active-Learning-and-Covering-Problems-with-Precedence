\begin{algorithm}[h]
\DontPrintSemicolon
\LinesNumbered
\caption{The greedy algorithm for MDPCS}
\label{alg:MDPCS}
\SetKwProg{Proc}{procedure}{}{}
\Proc{$\ProcedureMDPCSGreedy(\mathcal{G}, \preceq, R)$}{
$\mathcal{A} \gets \mathcal{G}$\;

\ForEach{$S \in \mathcal{G}$}{
\If{$\Delta\br{\closure{S}, R} > \Delta\br{\mathcal{A}, R}$}{
$\mathcal{A} \gets \closure{S}$\;
}
}
\Return $\mathcal{A}$\;
}
\end{algorithm}
