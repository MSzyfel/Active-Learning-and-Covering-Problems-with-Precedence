\SetKwRepeat{Do}{do}{while}%

\begin{algorithm}[h]
\DontPrintSemicolon
\LinesNumbered

\caption{A generic approximation algorithm for $\ProblemPCACDT$ and $\ProblemPCWCDT$.}
\label{ea:alg:worstDecisionTree}
\SetKwProg{Proc}{procedure}{}{}
\Proc{\ProcDecisionTree$(\cH, \cT, \preceq, \blackBox)$}{
\If{$\spr{\cH} = 1$}{
    \Return the trivial decision tree with a single leaf corresponding to the only hypothesis in $\cH$.
}

$\cC \gets$ Call $\br{\alpha, \beta}$-approximation algorithm $\blackBox$ on instance $(\cU, \xi(\cT), \preceq_{\xi}, f=1/4)$.

$D\gets\br{\emptyset, \emptyset}$ and let $i\gets 1$.

\Do{$i\leq|\cC|$ \textnormal{and there is no $\cH'$, $\cH'\in t_{i}$, such that $\spr{\cH'} > \frac{5\beta-1}{5\beta}\cdot \spr{\cH}$}}{

$t_i \gets$ the test associated with the $i$-th set in $\cC$.

Make $t_i$ to be a child of $t_{i-1}$ in $D$ (when $i>1$), or the root of $D$ otherwise (when $i=1$).

$\cH_i \gets\argmax_{\cH'\in t_{i}} \brc{\spr{\cH'}}$.

\ForEach{\textnormal{reply $\cH'$, $\cH'\in t_{i}$, such that $1\leq\spr{\cH'} \leq \frac{5\beta-1}{5\beta}\cdot \spr{\cH}$}}{
    $D_{\cH'} \gets$ \ProcDecisionTree$(\cH', \cT, \preceq, \blackBox)$.

    Hang $D_{\cH'}$ below $t_{i}$ in $D$, i.e., make the root of $D_{\cH'}$ to be a child of $t_i$.
}

Increment $i$.

% \If{\textnormal{$i\neq 1$ and there is no $\cH'$, $\cH'\in t_{i-1}\br{\cH_{i-1}}$, such that $\spr{\cH'} > \frac{5\beta-1}{5\beta}\cdot \spr{\cH}$}}{
%     \Return $D$.
% }

}
\Return $D$.
}
\end{algorithm}
